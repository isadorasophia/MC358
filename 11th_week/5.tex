\problem{5.}
Um grafo $ G $ é $ k $-regular se todo vértice de $ G $ tem grau $ k $.

    \begin{enumerate}[label=\arabic*.]
      \item Quais dos seguintes grafos são $ k $-regulares (para algum $ k $)?
        \begin{enumerate}[label=\arabic*.]
          \item grafos completos;
          \item ciclos;
          \item rodas;
          \item grafos bipartidos;
          \item $ n $-cubo;
          \item grafos bipartidos completos.
        \end{enumerate}
      \item Quantas arestas possui um grafo $ k $-regular com $ n $ vértices?
    \end{enumerate}

\solution
    \begin{enumerate}[label=\arabic*.]
      \item
        \begin{enumerate}[label=\arabic*.]
          \item Sim;
          \item Sim;
          \item Sim;
          \item Sim;
          \item Sim;
          \item Sim.
        \end{enumerate}
      \item O grafo possui $\cfrac{kn}{2}$ arestas.
    \end{enumerate}
