\problem{6.}
Resolva sem usar tabelas verdade.

  \begin{enumerate}[label=\arabic*.]
    \item Suponha que $ P \rightarrow Q $ e $ R \rightarrow \lnot Q $ são ambos verdadeiros. Prove que $ P \rightarrow \lnot R $.
    \item Suponha que $ P $ é verdadeiro. Prove que $ Q \rightarrow  \lnot (Q \rightarrow \lnot P) $ é verdadeiro.
  \end{enumerate}

\solution
    \begin{enumerate}[label=\arabic*.]
      \item Se $R \rightarrow \lnot Q$, então $\lnot R \lor \lnot Q = \lnot Q \lor \lnot R = Q \rightarrow \lnot R$. Ou seja, $P \rightarrow Q \rightarrow \lnot R$, que equivale a $P \rightarrow \lnot R$.
      \item $Q \rightarrow \lnot(Q \rightarrow \lnot P)$ equivale a $Q \rightarrow \lnot(\lnot Q \lor \lnot P) = Q \rightarrow (Q \land P) = (Q \rightarrow Q) \land P$. 

      Como $Q \rightarrow Q = \lnot Q \lor Q$ é sempre verdadeiro (por inferência), temos $Q \rightarrow \lnot(Q \rightarrow \lnot P) = $\textbf{T}$ \land P$. Ou seja, se $P$ é verdadeiro, a sentença é verdadeira.
    \end{enumerate}
