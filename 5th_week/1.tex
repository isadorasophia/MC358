\problem{1.}
Seja $ p $ a proposição `oito é um número par' e $ q $ a proposição `pirulito é doce'. Componha, em palavras:
    \begin{enumerate}[label=\arabic*.]
        \item a implicação $ p \rightarrow q $
        \item sua reciproca
        \item sua inversa
        \item sua contrapositiva
        \item sua negação
    \end{enumerate}

\solution
    \begin{enumerate}[label=\arabic*.]
        \item Se oito é um número par, então o pirulito é doce.
        \item Se o pirulito é doce, então oito é um número par.
        \item Se oito não é um número par, então o pirulito não é doce.
        \item Se o pirulito não é doce, então oito não é um número par.
        \item Existe um pirulito que não é doce se oito é um número par.
    \end{enumerate}
