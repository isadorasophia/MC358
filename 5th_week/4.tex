\problem{4.}
  Escreva a negação de cada proposição abaixo.

  \begin{enumerate}[label=\arabic*.]
    \item Todos os brasileiros são loucos.
    \item Todos os homens são honestos ou algum(s) homens são ladrões.
    \item Há pelo menos uma pessoa no mundo que está sempre feliz.
    \item Se $ x $ é menor que 10, então existe um $ y $ tal que $ x^2 + y^2 - 100 $ é positivo.
    \item Os lados correspondentes de dois triângulos são iguais apenas se os triângulos são congruentes.
    \item Para cada homem existe pelo menos uma mulher que o despreza.
  \end{enumerate}

\solution
  \begin{enumerate}[label=\arabic*.]
    \item Existe um brasileiro que não é louco.
    \item Existe homem não honesto ou todos os homens não são ladrões.
    \item Todas as pessoas no mundo não estão sempre felizes.
    \item Existe um $x$ menor que 10 e nenhum $y$ tal que $ x^2 + y^2 - 100 $ é positivo.
    \item Existe dois triângulos com lados correspondes iguais e que não sejam congruentes.
    \item Existe um homem e uma mulher que não o despreza.
  \end{enumerate}
